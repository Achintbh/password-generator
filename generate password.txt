def generate_password():
    import random

    # Definees  character sets:-
    lower_chars = list('abcdefghijklmnopqrstuvwxyz')
    upper_chars = list('ABCDEFGHIJKLMNOPQRSTUVWXYZ')
    digit_chars = list('0123456789')
    symbol_chars = list('!@#$%&*')

    print("PASSWORD CREATOR - Beginner Edition")
    print("Let's make a strong password for you!")
    #giving the inputs here:
    try:
        num_letters = int(input("How many letters do you want in your password? "))
        num_digits = int(input("How many numbers would you like? "))
        num_symbols = int(input("How many symbols would you prefer? "))
    except ValueError:
        print("Please enter a valid number!")
        return

    print("Should the letters be 'mixed' case, only 'lower', or only 'upper'? (Enter: mixed/lower/upper)")
    case_preference = input().strip().lower()

    password_chars = []
    #using a big chain of elseif if for else:
    if case_preference == "lower":
        for _ in range(num_letters):
            password_chars.append(random.choice(lower_chars))
    elif case_preference == "upper":
        for _ in range(num_letters):
            password_chars.append(random.choice(upper_chars))
    else:
        for _ in range(num_letters):
            if random.randint(0, 1) == 0:
                password_chars.append(random.choice(lower_chars))
            else:
                password_chars.append(random.choice(upper_chars))

    for _ in range(num_digits):
        password_chars.append(random.choice(digit_chars))

    for _ in range(num_symbols):
        password_chars.append(random.choice(symbol_chars))

    shuffle_choice = input("Do you want the password to be mixed up? (yes/no) ").strip().lower()
    if shuffle_choice == "yes":
        random.shuffle(password_chars)
    #givingg the final result:
    password_final = "".join(password_chars)
    print(f"Here is your generated password: {password_final}")
